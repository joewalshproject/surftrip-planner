    <script>

        function findSurfTrips() {

            let budget = document.getElementById("budget").value;

            let waveSize = document.getElementById("waveSize").value;

            let waterTemp = document.getElementById("waterTemp").value;

            let duration = document.getElementById("duration").value;

            

            let destinations = [

                { name: "Bali", waveSize: "shoulder", waterTemp: "75_plus", budget: 1500 },

                { name: "Hawaii", waveSize: "overhead", waterTemp: "75_plus", budget: 2000 },

                { name: "Portugal", waveSize: "head", waterTemp: "65_plus", budget: 1200 },

                { name: "Mexico", waveSize: "waist", waterTemp: "75_plus", budget: 1000 }

            ];

            

            let filtered = destinations.filter(d => d.waveSize === waveSize && d.waterTemp === waterTemp && d.budget <= budget);

            

            let resultsDiv = document.getElementById("results");

            resultsDiv.innerHTML = filtered.length ? filtered.map(d => `<p>${d.name} - ${duration} Days</p>`).join('') : "No results found.";

        }

    </script>
